Model

Model Model Equation
Normal Linear Model \(\lambda_n=exp(\beta_0+\beta_1x_{n1}+\cdots +\beta_{M}x_{nM})\)
\(p(y_n | \beta,\sigma) \sim Poiss(y_n|\lambda_n)\)
This Model \(\lambda_n=\beta_0+\beta_1Temp_n+\beta_2Holiday_n\)
\(p(Sales_n | \beta) \sim Poiss(Sales_n | \lambda_n)\)
Simulation \(\lambda_n=-2+0.2Temp_n+0.5Holiday_n\)
\(Sales_n \sim Poiss(Sales_n | \lambda_n)\)

Stan

Stan File

data {
  int N;
  int y[N];
  vector[N] x1;
  vector[N] x2;
  
  int N_hat;
  vector[N_hat] x_hat;
}

parameters {
  vector[3] b;
}

transformed parameters{
  vector[N] lambda;
  lambda = exp(b[1] + b[2]*x1 + b[3]*x2);
}

model{
  y ~ poisson(lambda);
}

generated quantities{
  vector[N_hat] lambda_hat1;
  vector[N_hat] lambda_hat0;
  int y_hat1[N_hat];
  int y_hat0[N_hat];
  
  for(n in 1:N_hat){
    lambda_hat1[n] = exp(b[1] + b[2]*x_hat[n] + b[3]);
    lambda_hat0[n] = exp(b[1] + b[2]*x_hat[n]);
    y_hat1[n] = poisson_rng(lambda_hat1[n]);
    y_hat0[n] = poisson_rng(lambda_hat0[n]);
  }
}

R for Stan

x_hat <- seq(min(data$temp), max(data$temp))
data_list <- list(
  N = nrow(data), y = data$sales,
  x1 = data$temp, x2 = data$holiday,
  
  N_hat = length(x_hat), x_hat = x_hat
)

mcmc_result <- stan(
  file="PoissonLinearModel.stan",
  data=data_list,
  seed=10,
  iter = 2000, warmup = 200, chains = 3, thin=1
)
