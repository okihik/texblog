Purpose

Bayesian estimation of the parameter p of the Bernoulli distribution will be implemented using R. For the prior distribution, a beta distribution will be specified, and the posterior distribution will be obtained. We will implement this from data generation, so it can be reproduced by copy and paste.

TOC

  1. Library
  2. True Distribution
  3. Prior Distribution
  4. Data Generation
  5. Posterior Distribution

Library

library(ggplot2)
set.seed(1)

True Distribution

x <- 0:1
p <- 0.8
prob <- dbinom(x, 1, p)

ggplot() + 
  geom_bar(aes(x, prob), stat ="identity") +
  labs(x="x", y="prob", title="Bernoulli Distribution")+
  theme_classic(base_family = "") +
  theme(text = element_text(size = 24))

Prior Distribution

a_pre <- 2
b_pre <- 2
x <- seq(0,1, 0.01)
p_prob_pre <- dbeta(x, a_pre, b_pre)

plot <- ggplot()+
  geom_line(aes(x=x, y=p_prob_pre))+
  labs(x="p", y="prob", title="Beta Distribution")+
  theme_classic(base_family = "") +
  theme(text = element_text(size = 24))
plot

Data Generation

data <- rbinom(30, 1, p)
N <- length(data)

Posterior Distribution

a_pos <- sum(data) + a_pre
b_pos <- N - sum(data) + b_pre
x <- seq(0,1, 0.01)
p_prob_pos <- dbeta(x, a_pos, b_pos)

plot + geom_line(aes(x=x, y=p_prob_pos), col="blue")
