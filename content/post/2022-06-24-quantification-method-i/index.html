Library

library(dplyr)
library(ggplot2)

Data Generation

set.seed(100)
N <- 30
lectures <- c(rep("lecture A", N), rep("lecture B", N), rep("lecture C", N))
e1 <- rnorm(N, 0, 10)
e2 <- rnorm(N, 0, 10)
e3 <- rnorm(N, 0, 10)
data <- data.frame(lectures) %>% 
mutate(test=ifelse(lectures=="lecture A", 50+e1 , ifelse(lectures=="lecture B", 30+e2, 80+e3))) %>%
mutate(test=round(test))
data %>% sample_n(5)
##    lectures test
## 1 lecture C   82
## 2 lecture C   68
## 3 lecture A   46
## 4 lecture A   43
## 5 lecture B   29
